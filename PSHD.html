What is PSHD Box?

Story of origin

One day John was playing mope.io as a Land monster on his laptop until a potted plant came along. The plant beat up John until he was lying down on the ground, moaning (the plant later died of clinical depression).

After 10 minutes of contemplating his life decisions, John wondered: “What if guns were legal?”, but God said

The word moist sounded so appealing to John so he thought about moist objects. Moist cake. Moist feet. Moist soil. Because of this thought John suddenly realised that the Wikipedia article about Mark Zuckerberg contained 6,265 words. He also realised that moist soil was a key component to keeping a plant alive. Being the reasonable unspecified male child he is, John realised the threats of the plant legion and decided to make a device that would instead keep a plant alive.

Code (Written with the Arduino IDE)

#include LiquidCrystal.h

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);
const int soilPin = A0;
const int soilPower = 7;
/*instead of 5V or 3.3V pins, we use digital pins to prevent corrosion of the soil*/
int moisture = 0;
int newMoisture = 0;

void setup() {
pinMode(soilPower, OUTPUT);
digitalWrite(soilPower, LOW);
lcd.begin(16,2);
lcd.clear();
}

void loop() { lcd.clear();
for(int i = 1; i<=30; i++){
lcd.setCursor(0,0);
lcd.print("Soil Moisture:");
lcd.setCursor(0,1);
readSoil();
lcd.print(newMoisture);
lcd.print("%");
delay(30000);
lcd.clear();
lcd.print("Flowers:30%-50% Fruits:40%-60%");
lcd.setCursor(0,1);
lcd.print("Trees:21%-50% Vgtables:41%-80%");
delay(11400);
for(int j = 1;j < 17;j++){
lcd.scrollDisplayLeft();
delay(200);
}
delay(11400);
for(int j = 1;j < 17;j++){
lcd.scrollDisplayRight();
delay(200);
}
delay(800);
lcd.clear();
}
}
int readSoil(){
digitalWrite(soilPower, HIGH);
delay(10);
moisture = analogRead(soilPin);
digitalWrite(soilPower, LOW);
newMoisture = map(moisture,0,1023,0,100);
return newMoisture;
}
